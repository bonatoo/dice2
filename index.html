// Get references to the dice image, roll button, and stats table const dice = document.getElementById("dice"); const rollButton = document.getElementById("rollButton"); const statsTable = document.getElementById("statsTable"); const resetButton = document.getElementById("resetButton"); // Add event listeners to the roll and reset buttons rollButton.addEventListener("click", rollDice); resetButton.addEventListener("click", resetStats); function rollDice() { // Generate a random number using the crypto object const array = new Uint32Array(1); window.crypto.getRandomValues(array); const randomNumber = array[0] % 6 + 1; // Update the dice image and statistics table dice.src = "dice" + randomNumber + ".png"; document.getElementById("rolls" + randomNumber).innerHTML++; // Add text that displays the number rolled next to the dice image const diceRollText = document.createTextNode("Dice rolled " + randomNumber); dice.parentNode.insertBefore(diceRollText, dice.nextSibling); } function resetStats() { // Reset all the values in the statistics table to 0 for (let i = 1; i <= 6; i++) { document.getElementById("rolls" + i).innerHTML = 0; } }
